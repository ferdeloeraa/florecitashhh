.container { text-align: center; margin-top: 50px; } .flower { width: 50px; height: 50px; background-color: yellow; border-radius: 50%; display: inline-block; margin: 10px; } document.addEventListener('DOMContentLoaded', function() { const flowersContainer = document.getElementById('flowers'); // Número de flores a mostrar const numFlowers = 10; // Generar las flores dinámicamente for (let i = 0; i < numFlowers; i++) { const flower = document.createElement('div'); flower.className = 'flower'; flowersContainer.appendChild(flower); } });
